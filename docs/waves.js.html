    waves.push(new Wave([

    {"enemy":new Tier1Enemy(path),"time":160},
  ]));
  // wave 3
  // 3 Tier 1s
  // 1 Tier 2 Bases
  // 2 Tier 2 Tanks
  waves.push(new Wave([
    {"enemy":new Tier1Enemy(path),"time":5},
    {"enemy":new Tier1Enemy(path),"time":15},
    {"enemy":new Tier1Enemy(path),"time":25},
